class MemoryDumper {
private:
    HANDLE hProcess;
    DWORD_PTR baseAddress;
    DWORD processID;

    std::vector<BYTE> HexPatternToBytes(const std::wstring& pattern) {
        std::vector<BYTE> bytes;
        std::wstring byteStr;

        for (size_t i = 0; i < pattern.length(); i++) {
            if (pattern[i] == L' ') continue;
            if (pattern[i] == L'?') {
                bytes.push_back(0x00);
                if (i + 1 < pattern.length() && pattern[i + 1] == L'?') i++;
            }
            else {
                byteStr += pattern[i];
                if (byteStr.length() == 2) {
                    bytes.push_back(static_cast<BYTE>(std::wcstoul(byteStr.c_str(), nullptr, 16)));
                    byteStr.clear();
                }
            }
        }
        return bytes;
    }

    std::wstring CreateMask(const std::vector<BYTE>& pattern) {
        std::wstring mask;
        for (BYTE b : pattern) {
            mask += (b == 0x00) ? L'?' : L'x';
        }
        return mask;
    }

    bool PatternMatch(const BYTE* data, const BYTE* pattern, const std::wstring& mask) {
        for (size_t i = 0; i < mask.length(); i++) {
            if (mask[i] == L'x' && data[i] != pattern[i]) return false;
        }
        return true;
    }

    DWORD_PTR FindPattern(const std::vector<BYTE>& pattern, const std::wstring& mask);

    bool GetRobloxBaseAddress() {
        HMODULE hMods[1024];
        DWORD cbNeeded;

        if (EnumProcessModules(hProcess, hMods, sizeof(hMods), &cbNeeded)) {
            baseAddress = (DWORD_PTR)hMods[0];
            return true;
        }
        return false;
    }

public:
    MemoryDumper(DWORD pid) : hProcess(NULL), baseAddress(0), processID(pid) {
        hProcess = OpenProcess(PROCESS_ALL_ACCESS, FALSE, pid);
        if (hProcess) {
            GetRobloxBaseAddress();
        }
    }

    ~MemoryDumper() {
        if (hProcess) CloseHandle(hProcess);
    }

    struct PatternResult {
        std::wstring name;
        DWORD_PTR address;
        std::vector<BYTE> data;
    };

    std::vector<PatternResult> DumpPatterns(const std::vector<std::pair<std::wstring, std::wstring>>& patterns) {
        std::vector<PatternResult> results;

        for (const auto& pattern : patterns) {
            std::vector<BYTE> patternBytes = HexPatternToBytes(pattern.second);
            std::wstring mask = CreateMask(patternBytes);

            DWORD_PTR address = FindPattern(patternBytes, mask);
            if (address) {
                PatternResult result;
                result.name = pattern.first;
                result.address = address;

                // ДАМПИМ 256 БАЙТ ВОКРУГ НАЙДЕННОГО АДРЕСА
                result.data.resize(256);
                ReadProcessMemory(hProcess, (LPCVOID)address, result.data.data(), 256, NULL);

                results.push_back(result);
            }
        }
        return results;
    }

    void SaveDumpToFile(const std::wstring& filename, const std::vector<PatternResult>& results) {
        std::wofstream file(filename, std::ios::binary);
        if (file.is_open()) {
            file << L"ROBLOX MEMORY DUMP - PATTERN BASED\n";
            file << L"Process ID: " << processID << L"\n";
            file << L"Base Address: 0x" << std::hex << baseAddress << L"\n\n";

            for (const auto& result : results) {
                file << L"[" << result.name << L"]\n";
                file << L"Address: 0x" << std::hex << result.address << L"\n";
                file << L"Data: ";

                for (size_t i = 0; i < result.data.size(); i++) {
                    if (i % 16 == 0) file << L"\n";
                    file << std::hex << static_cast<int>(result.data[i]) << L" ";
                }
                file << L"\n\n";
            }
            file.close();
        }
    }
};

DWORD GetRobloxProcessID() {
    HANDLE hSnapshot = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);
    PROCESSENTRY32 pe32;
    pe32.dwSize = sizeof(PROCESSENTRY32);

    if (Process32First(hSnapshot, &pe32)) {
        do {
            if (wcscmp(pe32.szExeFile, L"RobloxPlayerBeta.exe") == 0) {
                CloseHandle(hSnapshot);
                return pe32.th32ProcessID;
            }
        } while (Process32Next(hSnapshot, &pe32));
    }
    CloseHandle(hSnapshot);
    return 0;
}

inline DWORD_PTR MemoryDumper::FindPattern(const std::vector<BYTE>& pattern, const std::wstring& mask) {
    MEMORY_BASIC_INFORMATION mbi;
    DWORD_PTR currentAddress = 0;

    while (VirtualQueryEx(hProcess, (LPCVOID)currentAddress, &mbi, sizeof(mbi))) {
        if (mbi.State == MEM_COMMIT &&
            (mbi.Protect == PAGE_READONLY ||
                mbi.Protect == PAGE_READWRITE ||
                mbi.Protect == PAGE_EXECUTE_READ ||
                mbi.Protect == PAGE_EXECUTE_READWRITE)) {
            SIZE_T bufferSize = mbi.RegionSize;
            BYTE* buffer = new BYTE[bufferSize];

            if (ReadProcessMemory(hProcess, mbi.BaseAddress, buffer, bufferSize, NULL)) {
                for (SIZE_T i = 0; i <= bufferSize - pattern.size(); i++) {
                    if (PatternMatch(buffer + i, pattern.data(), mask)) {
                        delete[] buffer;
                        return (DWORD_PTR)mbi.BaseAddress + i;
                    }
                }
            }
            delete[] buffer;
        }
        currentAddress = (DWORD_PTR)mbi.BaseAddress + mbi.RegionSize;
    }
    return 0;
}
